                  </select>
                </div>
              </div>
            )}
 
            {activeOrderType === 'limit' && (
              <div className="space-y-4">
                <div>
                  <label className="block text-white mb-2">Limit Price</label>
                  <input
                    type="number"
                    step="0.01"
                    value={orderConfig.limit.price}
